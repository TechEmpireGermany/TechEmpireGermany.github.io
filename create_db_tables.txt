CREATE TABLE support_team_applications (
id INT AUTO_INCREMENT,
userid BIGINT UNSIGNED,
username VARCHAR(420),
q1 VARCHAR(32),
q2 VARCHAR(8),
q3 VARCHAR(64),
q4 VARCHAR(8),
q5 VARCHAR(8),
time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id)
);

CREATE TABLE moderator_applications (
id INT AUTO_INCREMENT,
userid BIGINT UNSIGNED,
username VARCHAR(420),
q1 VARCHAR(32),
q2 VARCHAR(16),
q3 VARCHAR(128),
q4 VARCHAR(128),
q5 VARCHAR(16),
q6 VARCHAR(16),
q7 VARCHAR(16),
time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id)
);

CREATE TABLE appeals (
id INT AUTO_INCREMENT,
userid BIGINT UNSIGNED,
username VARCHAR(420),
email VARCHAR(420),
q1 TEXT,
q2 TEXT,
time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id)
);
